\chapter*{List of Symbols and Notation}
\addcontentsline{toc}{chapter}{List of Symbols and Notation}

The following tables define the principal symbols and
notation used throughout this thesis.
Each symbol is listed with its definition and the section
in which it is first introduced.

%% ─── Federated Learning ────────────────────────────────────

  \hline
  \textbf{Symbol} & \textbf{Definition} & \textbf{Section} \\
  \hline
  $K$
    & Number of hospital clients in the federation
    & §\ref{sec:fl-clients} \\

    & Total number of communication rounds
    & §\ref{sec:round-config} \\

    & Number of local SGD epochs per round per client
    & §\ref{sec:round-config} \\

    & §\ref{sec:stage3} \\

      heterogeneous distributions
    & §\ref{sec:noniid-partition} \\

    & Number of training samples at client $k$
    & §\ref{eq:fedavg-ch3} \\

    & Total training samples: $n = \sum_{k=1}^K n_k$
    & §\ref{eq:fedavg-ch3} \\
  $\mathbf{w}^{(t)}$
    & Global model parameters at round $t$
    & §\ref{sec:fedavg-baseline} \\
  $\mathbf{w}_k^{(t)}$
    & Local model parameters at client $k$ after round $t$
    & §\ref{sec:fedavg-baseline} \\
  $\mathbf{c}$
    & SCAFFOLD server control variate
    & §\ref{sec:scaffold} \\
  $\mathbf{c}_k$
    & SCAFFOLD client control variate for client $k$
    & §\ref{sec:scaffold} \\
  $\beta$
    & Server momentum coefficient (Momentum-FedAvg)
    & §\ref{sec:fl-optimised} \\
  $\mathbf{v}^{(t)}$
    & Server momentum velocity vector at round $t$
    & §\ref{eq:momentum-fedavg} \\

    & Server learning rate (global update step size)
    & §\ref{eq:momentum-fedavg} \\

\end{center}

%% ─── Model Architecture ────────────────────────────────────
\section*{Model Architecture}
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{p{2.8cm} p{9.5cm} p{1.6cm}}
  \hline
  \textbf{Symbol} & \textbf{Definition} & \textbf{Section} \\
  \hline
  $\Psi_\theta$
    & ConvNeXt-Nano backbone mapping input slice to
      feature vector; $\theta$ denotes all backbone
      parameters
    & §\ref{sec:convnext-backbone} \\
  $d$
    & Backbone output feature dimension ($d = 640$)
    & §\ref{sec:convnext-backbone} \\
  $\mathbf{f}_k$
    & Backbone feature vector for axial slice $k$:
      $\mathbf{f}_k \in \mathbb{R}^{640}$
    & §\ref{eq:proj} \\
  $\mathbf{h}_k$
    & Projected feature vector after MIL projection:
      $\mathbf{h}_k \in \mathbb{R}^{256}$
    & §\ref{eq:proj} \\
  $a_k$
    & Raw gated attention score for slice $k$
    & §\ref{eq:gated-attention} \\
  $\hat{a}_k$
    & Softmax-normalised attention weight for slice $k$
    & §\ref{eq:attn-softmax} \\
  $\mathbf{z}$
    & Volume-level MIL embedding:
      $\mathbf{z} = \sum_k \hat{a}_k \mathbf{h}_k
      \in \mathbb{R}^{256}$
    & §\ref{eq:mil-aggregate} \\
  $V,\, U$
    & Gated attention weight matrices:
      $V, U \in \mathbb{R}^{128 \times 256}$
    & §\ref{eq:gated-attention} \\
  $\mathbf{w}_{\mathrm{attn}}$
    & Gated attention weight vector:
      $\mathbf{w}_{\mathrm{attn}} \in \mathbb{R}^{128}$
    & §\ref{eq:gated-attention} \\
  $C$
    & Number of output classes
      ($C=4$ for 4-class; $C=2$ for binary)
    & §\ref{sec:dataset-description} \\
  $S$
    & Number of axial slices per volume ($S = 64$)
    & §\ref{sec:slice-extraction} \\
  GELU
    & Gaussian Error Linear Unit:
      $\mathrm{GELU}(x) = x\,\Phi(x)$, where $\Phi$ is
      the standard normal CDF.
      Used in the MIL projection layer.
    & §\ref{eq:proj} \\
  \hline
\end{tabular}
\end{center}

%% ─── FedSCRT Theoretical Framework ────────────────────────
\section*{FedSCRT Theoretical Framework}
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{p{2.8cm} p{9.5cm} p{1.6cm}}
  \hline
  \textbf{Symbol} & \textbf{Definition} & \textbf{Section} \\
  \hline
  $\mathcal{P}_k(c)$
    & Class prior at client $k$: probability that a
      randomly drawn sample from $\mathcal{D}_k$ belongs
      to class $c$. Varies across clients under non-IID.
    & §\ref{thm:fedscrt} \\
  $\mathcal{P}_k(x \mid c)$
    & Class-conditional feature distribution at client $k$:
      distribution of input features given class label $c$.
      Assumed approximately equal across clients (covariate
      shift assumption).
    & §\ref{thm:fedscrt} \\
  $\mathcal{P}(x \mid c)$
    & Global class-conditional feature distribution
    & §\ref{thm:fedscrt} \\
  $\mathcal{L}_{\mathrm{bal}}(\mathbf{w})$
    & Global balanced classification objective:
      equal weight per class
    & §\ref{eq:balanced-objective} \\
  $\mathcal{L}_k(\mathbf{w})$
    & Local imbalanced objective at client $k$:
      weighted by local class prior $\mathcal{P}_k(c)$
    & §\ref{eq:local-objective} \\
  $\mathcal{L}_k^{\mathrm{bal}}(\mathbf{w})$
    & Local objective under class-balanced sampling
      at client $k$: equal weight per class regardless
      of $\mathcal{P}_k(c)$
    & §\ref{eq:fedscrt-unbiased} \\
  $m$
    & Number of samples drawn per class per client
      during balanced sampling
    & §\ref{eq:per-class-grad} \\
  $\mathbf{w}_k^{\mathrm{cRT}}$
    & Local cRT head parameters at client $k$ after
      $T$ balanced training epochs
    & §\ref{cor:fedscrt-agg} \\
  $\mathbf{w}_{\mathrm{global}}$
    & Aggregated FedSCRT global head
    & §\ref{cor:fedscrt-agg} \\
  \hline
\end{tabular}
\end{center}
